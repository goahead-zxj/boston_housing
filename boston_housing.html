def performance_metric(y_true, y_predict):
    """"""
    
    length = len(y_true)
    RSS = 0
    TSS = 0
    y_true_mean = np.mean(y_true)
    for item in range(length):
        rss = (y_true[item]-y_predict[item])**2
        RSS += rss
        tss = (y_true[item]-y_true_mean)**2
        TSS += tss
    
    score = 1-RSS/TSS

    return score
